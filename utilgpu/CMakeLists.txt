option(BUILD_CPP "Build cpp sublibrary" ON)
option(BUILD_QT  "Build qt sublibrary (requires Qt)" OFF)
option(BUILD_GL  "Build gl sublibrary (requires glbinding)" OFF)
option(BUILD_ALL "Build all sublibraries" OFF)

if(BUILD_CPP OR BUILD_ALL)
    add_subdirectory(cpp)
endif()
if(BUILD_QT OR BUILD_ALL)
    add_subdirectory(qt)
endif()
if(BUILD_GL OR BUILD_ALL)
    add_subdirectory(gl)
endif()

install (DIRECTORY ${CMAKE_CURRENT_LIST_DIR} DESTINATION include FILES_MATCHING PATTERN "*.h")
